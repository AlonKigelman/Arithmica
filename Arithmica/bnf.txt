<assignment> ::= <identifier> "=" <expression>

<expression> ::= <operator> (<operand> | <expression>)*

<operator> ::= "+" | "-" | "*" | "/"

<operand> ::= <number> | <identifier>

<identifier> ::= <letter> (<letter> | <digit>)*

<number> ::= <digit>+

<letter> ::= "a" | "b" | ... | "z" | "A" | "B" | ... | "Z"

<digit> ::= "0" | "1" | ... | "9"
